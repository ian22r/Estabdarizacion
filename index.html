function calcCostoReceta(receta) {

  var total = 0;

  for (var i = 0; i < receta.ingredientes.length; i++) {

    var ri = receta.ingredientes[i];

    var ing = getIngrediente(ri.ingredienteId);

    if (!ing) continue;

    var mermaFactor = 1 + (ri.merma || 0) / 100;

    total += ing.precio * ri.cantidad * mermaFactor;

  }

  var imp = total * ((receta.imprevistos || 10) / 100);

  var costoTotal = total + imp;

  var costoPorcion = costoTotal / (receta.porciones || 1);

  var precioVenta = costoPorcion * (1 + (receta.margenVenta || 300) / 100);

  return { total: total, imprevistos: imp, costoTotal: costoTotal, costoPorcion: costoPorcion, precioVenta: precioVenta, margen: receta.margenVenta || 300 };

}

  else if (sort === 'margen') list.sort(function(a, b) { return b.margenVenta - a.margenVenta; });

  }

  tbody.innerHTML = html;

function renderCostos() {

  for (var i = 0; i < recetas.length; i++) {

  var rankHtml = '';

    cats[rc.clasificacion].count++;

  }

  var catData = Object.keys(cats).map(function(key) { return { cat: key, avg: cats[key].total / cats[key].count, count: cats[key].count }; });

  catData.sort(function(a, b) { return b.avg - a.avg; });

function simularPrecio() {

  var id = parseInt(document.getElementById('sim-receta').value);

  var margen = parseFloat(document.getElementById('sim-margen').value) || 300;

  var imp = parseFloat(document.getElementById('sim-imprevistos').value) || 10;

  if (!id) { res.style.display = 'none'; return; }

  var r = null;

  for (var i = 0; i < recetas.length; i++) { if (recetas[i].id === id) { r = recetas[i]; break; } }

  if (!r) return;

  for (var j = 0; j < r.ingredientes.length; j++) {

}
